# nosc=22.0
# #map_radius = 2.8 # arcminutes
# map_radius = 1.5 # arcminutes
# #radial_osc_period = 25.0 # seconds
# radial_osc_period = 28.0 # seconds
# #n_scans = 5 # split 22 oscillations over 5 scans
# n_scans = 4 # split 22 oscillations over 4 scans
# daisyScanDuration = nosc * radial_osc_period / n_scans
